153 Connection::parseElement(TiXmlElement* elem) {
154  DEBUG(4, Connection,
155  "In parseElement.")
156 
157  if (elem->Attribute("id"))
158  m_connectionId = elem->Attribute("id");
159  DEBUG(5, Connection, "Processing connectionId " << m_connectionId);
160 
161  TiXmlElement * port = elem->FirstChild()->ToElement();
162 
163  if (!port)
164  throw CF::ApplicationFactory::CreateApplicationError(CF::CFNOTSET,
165  "No ports for connectioninterface.");
166 
167  //check if this connection is external. if there are two the same type port,
168  //this connection is for external port
169  int theSameUseNum = 0;
170  int theSameProvideNum = 0;
171  for (; port; port = port->NextSiblingElement()) {
172  std::string nodeName = port->Value();
173 
174  if (strcmp(nodeName.c_str(), "usesport") == 0) {
175  theSameUseNum++;
176  if (theSameUseNum >= 2) {
177  m_ifExternalUseConnect = true;
178  break;
179  }
180  } else if (strcmp(nodeName.c_str(), "providesport") == 0) {
181  theSameProvideNum++;
182  if (theSameProvideNum >= 2) {
183  m_ifExternalProvideConnect = true;
184  break;
185  }
186  }
187  } //for()
188 
189  TiXmlElement * _port = elem->FirstChild()->ToElement();
190  //begin to parse port, divide into three situation
191  if (m_ifExternalUseConnect == true) {
192  DEBUG(5, Connection::parseElement, "m_ifExternalUseConnect is true")
193  for (; _port; _port = _port->NextSiblingElement()) {
194  std::string nodeName = _port->Value();
195 
196  if (strcmp(nodeName.c_str(), "usesport") == 0) {
197  DEBUG(5, Connection, "Found usesport and _port is " << _port);
198  m_externalUsesPorts.push_back(new UsesPort(_port));
199 
200  } else {
201  DEBUG(2, Connection, "find Unknown port type when parse \
202  external uses port")
203  }
204  }
205 
206  DEBUG(5, Connection, "externalUsesPort size = " <<
207  m_externalUsesPorts.size());
208  } else if ( m_ifExternalProvideConnect == true ) {
209  DEBUG(5, Connection::parseElement,
210  "m_ifExternalProvideConnect is true")
211 
212  for (; _port; _port = _port->NextSiblingElement()) {
213  std::string nodeName = _port->Value();
214 
215  if (0 == strcmp(nodeName.c_str(), "providesport")) {
216  DEBUG(5, Connection,
217  "Found providesport and _port is " << _port);
218  m_externalProvidesPorts.push_back(new ProvidesPort(_port));
219  } else {
220  DEBUG(2, Connection,
221  "find Unknown port type when parse external provide port")
222  }
223  }
224  } else {
225  DEBUG(5, connect::parseElement, "parse internal port")
226 
227  for (; _port; _port = _port->NextSiblingElement()) {
228  std::string nodeName = _port->Value();
229 
230  DEBUG(5, connect::parseElement, "_port->Value " << _port->Value())
231 
232  if (strcmp(nodeName.c_str(), "usesport") == 0) {
233  DEBUG(5, Connection, "Found usesport.");
234  m_usesPort = new UsesPort(_port);
235  m_ifUsesPort = true;
236 
237  } else if (strcmp(nodeName.c_str(), "providesport") == 0) {
238  DEBUG(5, Connection, "Found providesport");
239  m_providesPort = new ProvidesPort(_port);
240  m_ifProvidesPort = true;
241  } else {
242  DEBUG(2, Connection,
243  "find Unknown port type when parse uses and provide port")
244  }
245  }
246  }
247 
248  DEBUG(4, Connection, "Finished processing connection.")
249 }
